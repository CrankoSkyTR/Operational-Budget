-- Step 1: Create tables
create table if not exists bpo_state (
  id text primary key default 'main',
  data jsonb not null,
  updated_at timestamptz default now()
);
create table if not exists bpo_users (
  id uuid references auth.users primary key,
  name text,
  role text default 'Sr. Manager',
  assigned_project text default null,
  created_at timestamptz default now()
);

-- Step 2: Enable RLS
alter table bpo_state enable row level security;
alter table bpo_users enable row level security;

-- Step 3: Drop old policies first (clean slate)
drop policy if exists "Auth users full access" on bpo_state;
drop policy if exists "Own profile" on bpo_users;
drop policy if exists "Admins manage all" on bpo_users;
drop policy if exists "Sr Manager manage all" on bpo_users;
drop policy if exists "All authenticated users read bpo_users" on bpo_users;

-- Step 4: bpo_state тАФ full access for logged in users
create policy "Auth users full access" on bpo_state
  for all to authenticated
  using (true) with check (true);

-- Step 5: bpo_users тАФ ALL authenticated users can read ALL rows
-- (needed so users can see their own profile)
create policy "Authenticated read all" on bpo_users
  for select to authenticated
  using (true);

-- Step 6: bpo_users тАФ only Sr. Manager can insert/update/delete
create policy "Sr Manager write" on bpo_users
  for all to authenticated
  using (
    exists (
      select 1 from bpo_users u
      where u.id = auth.uid()
      and u.role in ('Sr. Manager','Administrator')
    )
  )
  with check (true);

-- Step 7: Auto-create profile on signup
create or replace function public.handle_new_user()
returns trigger language plpgsql security definer as $$
begin
  insert into public.bpo_users(id, name, role)
  values (
    new.id,
    coalesce(new.raw_user_meta_data->>'name', split_part(new.email,'@',1)),
    'Sr. Manager'
  )
  on conflict (id) do nothing;
  return new;
end;$$;

drop trigger if exists on_auth_user_created on auth.users;
create trigger on_auth_user_created
  after insert on auth.users
  for each row execute procedure public.handle_new_user();
