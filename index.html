SQL — run once in Supabase SQL Editor 
-- Run this in Supabase SQL Editor (replaces all old policies)

-- Tables
create table if not exists bpo_state (
  id text primary key default 'main',
  data jsonb not null,
  updated_at timestamptz default now()
);
create table if not exists bpo_users (
  id uuid references auth.users primary key,
  name text, role text default 'Sr. Manager',
  assigned_project text default null,
  created_at timestamptz default now()
);

-- Enable RLS
alter table bpo_state enable row level security;
alter table bpo_users enable row level security;

-- Drop ALL old policies (prevents conflicts)
do $$ declare r record;
begin for r in select policyname from pg_policies
  where tablename in ('bpo_state','bpo_users') loop
  execute 'drop policy if exists "' || r.policyname || '" on ' ||
    case when r.policyname like '%state%' or
      exists(select 1 from pg_policies where tablename='bpo_state'
        and policyname=r.policyname)
      then 'bpo_state' else 'bpo_users' end;
end loop; end $$;

-- bpo_state: all authenticated users
drop policy if exists "Auth users full access" on bpo_state;
create policy "Auth users full access" on bpo_state
  for all to authenticated using (true) with check (true);

-- bpo_users: ALL authenticated users can READ (no recursion!)
drop policy if exists "Authenticated read all" on bpo_users;
create policy "Authenticated read all" on bpo_users
  for select to authenticated using (true);

-- ══ IMPORTANT: Run this in Supabase SQL Editor to fix recursion ══
-- Drop ALL existing bpo_users policies first, then recreate clean:
drop policy if exists "Authenticated read all" on bpo_users;
drop policy if exists "Own row update" on bpo_users;
drop policy if exists "Own row delete" on bpo_users;
drop policy if exists "Service insert" on bpo_users;
drop policy if exists "Sr Manager write" on bpo_users;
drop policy if exists "Sr Manager manage all" on bpo_users;
drop policy if exists "Own profile" on bpo_users;
drop policy if exists "Admins manage all" on bpo_users;

-- Allow ALL authenticated users to read (no self-join = no recursion)
create policy "Read all users" on bpo_users
  for select to authenticated using (true);

-- Allow users to update ONLY their own row
create policy "Update own row" on bpo_users
  for update to authenticated
  using (auth.uid() = id) with check (auth.uid() = id);

-- Allow insert for new signups (trigger uses security definer so no policy needed,
-- but client-side createUser also inserts — allow all authenticated)
create policy "Insert authenticated" on bpo_users
  for insert to authenticated with check (true);

-- bpo_users: each user can update their OWN row (name only)
drop policy if exists "Own row update" on bpo_users;
create policy "Own row update" on bpo_users
  for update to authenticated
  using (auth.uid() = id) with check (auth.uid() = id);

-- bpo_users: insert for service/trigger only (no client insert)
drop policy if exists "Service insert" on bpo_users;
create policy "Service insert" on bpo_users
  for insert to authenticated with check (true);

-- bpo_users: delete own row
drop policy if exists "Own row delete" on bpo_users;
create policy "Own row delete" on bpo_users
  for delete to authenticated using (auth.uid() = id);

-- Auto-create profile on signup
create or replace function public.handle_new_user()
returns trigger language plpgsql security definer as $$
begin
  insert into public.bpo_users(id,name,role)
  values(new.id,
    coalesce(new.raw_user_meta_data->>'name',split_part(new.email,'@',1)),
    'Sr. Manager')
  on conflict (id) do nothing;
  return new;
end;$$;
drop trigger if exists on_auth_user_created on auth.users;
create trigger on_auth_user_created
  after insert on auth.users for each row
  execute procedure public.handle_new_user();
